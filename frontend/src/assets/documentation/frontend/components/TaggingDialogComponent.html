import {Component, Inject} from '@angular/core';
import {MAT_DIALOG_DATA, MatDialogRef} from '@angular/material/dialog';
import {TranslateService} from '@ngx-translate/core';
import {IFile} from '../app/models/IFile.model';

export interface TaggingDialogData {
  status: string;
  progress: number;
  jobId: string;
  timeRemaining: number
}

@Component({
  selector: 'app-tagging-dialog',
  templateUrl: 'tagging.dialog.html',
})
export class TaggingDialogComponent {

  constructor(
    public dialogRef: MatDialogRef<TaggingDialogComponent>,
    public translate: TranslateService,
    @Inject(MAT_DIALOG_DATA) public data: TaggingDialogData) {}

  onDialogButtonClick(response: string): void {
    this.dialogRef.close(response);
  }
}

<h1 mat-dialog-title>{{'TAGGING' | translate}}</h1>
<div mat-dialog-content>
  {{data.status | translate}}
  <mat-progress-bar mode="determinate" value="{{data.progress}}"></mat-progress-bar>
  <span class="time-remaining" *ngIf="data.timeRemaining && data.timeRemaining !== -1 && data.status.toLowerCase() !== 'finished'">
    Time Remaining: {{data.timeRemaining.toFixed(1) + 2}}s
  </span>

</div>

<div mat-dialog-actions>
  <button mat-button color="primary" (click)="dialogRef.close()">{{'CLOSE' | translate}}</button>
  <button mat-button color="warn" [disabled]="data.status.toLowerCase() === 'finished'" (click)="onDialogButtonClick('cancel')">{{'CANCEL' | translate }}</button>
</div>
